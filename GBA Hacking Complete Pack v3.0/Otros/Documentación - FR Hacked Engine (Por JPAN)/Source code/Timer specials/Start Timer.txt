.align 2
.thumb

/*Special 0x46 Starts the timer*/

special_46: 	push {lr}
		ldr r0, timer_1
		mov r1, #0x0
		strb r1, [r0, #0x4]
		mov r1, #0x84
		strb r1, [r0, #0x6]
		mov r1, #0xc0
		lsl r1, r1, #0x8
		strh r1, [r0, #0x0]
		mov r1, #0x83
		strb r1, [r0, #0x2]
		pop {pc}
timer_1: .word 0x04000108
