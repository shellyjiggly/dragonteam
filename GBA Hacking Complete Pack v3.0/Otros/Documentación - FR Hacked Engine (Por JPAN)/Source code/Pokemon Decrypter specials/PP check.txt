.thumb
.align 2

/*Special 0x19 is the get PP special 
0x8005 is the attack slot
returns the pp value on that slot*/

Special_19:	push {r1-r4, lr}
		ldr r1, var_8005
		ldr r0, new_ATK_addr
		ldrh r1, [r1]
		cmp r1, #0x3
		bgt end_19
		ldr r0, [r0]
		add r0, #0x8
		add r0, r0, r1
		ldrb r0, [r0] 
end_19:		pop {r1-r4, pc}
.hword 0x0000
var_8005: .word 0x020370c2
new_ATK_addr: .word 0x0203f468
