.align 2
.thumb

/*Special 0x2b checks for the A and B keys. Returns
0x0 if none pressed
0x1 if A pressed
0x2 if B pressed
0x3 if both are pressed*/

special_2b:	push {r1, lr}
		ldr r0, key_reg
		ldrh r0 , [r0]
		mvn r0, r0
		mov r1, #0x3
		and r0, r1
		pop {r1, pc}
.hword 0x0000
key_reg: .word 0x04000130
