.align 2
.thumb
/*Special 0x2d checks for the start select buttons
0x0 if none is pressed
0x1 if select is pressed
0x2 if start is pressed
0x3 if both are pressed*/

special_2d:	push {r1, lr}
		ldr r0, key_reg2
		ldrh r0 , [r0]
		mvn r0, r0
		lsr r0, r0, #0x2
		mov r1, #0x3
		and r0, r1
		pop {r1, pc}
key_reg2: .word 0x04000130
